  <div className="text-center mb-12">
    <h2 className="text-3xl font-bold text-gray-900 dark:text-white mb-4">Quick Start</h2>
    <p className="text-lg text-gray-600 dark:text-gray-400">Get BeanQuick up and running in minutes</p>
  </div>

  <Steps>
    <Step title="Set up your environment">
      Clone the repository and install dependencies for both backend and frontend:
      
      ```bash
      # Clone the repository
      git clone https://github.com/SeboskiHub/bean-quick.git
      cd bean-quick
      
      # Set up backend
      cd back-end
      composer install
      cp .env.example .env
      php artisan key:generate
      
      # Set up frontend
      cd ../front-end
      npm install
      ```
    </Step>

    <Step title="Configure your database">
      Update your `.env` file with database credentials and run migrations:
      
      ```bash
      php artisan migrate
      ```
      
      <Accordion title="Sample database configuration">
      ```env
      DB_CONNECTION=mysql
      DB_HOST=127.0.0.1
      DB_PORT=3306
      DB_DATABASE=beanquick
      DB_USERNAME=root
      DB_PASSWORD=
      ```
      </Accordion>
    </Step>

    <Step title="Start the development servers">
      Launch both backend and frontend servers:
      
      ```bash
      # In back-end directory
      composer dev
      
      # This runs:
      # - Laravel development server (port 8000)
      # - Queue worker for background jobs
      # - Vite for asset compilation
      ```
      
      Visit `http://localhost:5173` to access the application.
    </Step>

    <Step title="Create your first admin account">
      Register a user account and manually set the role to admin in the database:
      
      ```sql
      UPDATE users SET role = 'admin' WHERE email = 'your-email@example.com';
      ```
      
      <Info>The first registered user should be promoted to admin to manage business applications.</Info>
    </Step>
  </Steps>
</div>
